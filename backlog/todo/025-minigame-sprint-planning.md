function evaluateSprint(selected: SprintTask[], scenario: SprintScenario): number {
  let score = 0
  const totalSP = selected.reduce((sum, t) => sum + t.storyPoints, 0)

  // 1. Critical –∑–∞–¥–∞—á–∏ –≤–∫–ª—é—á–µ–Ω—ã? (+200 –∑–∞ –∫–∞–∂–¥—É—é, -300 –∑–∞ –ø—Ä–æ–ø—É—â–µ–Ω–Ω—É—é)
  const criticalTasks = scenario.tasks.filter(t => t.priority === 'critical')
  for (const task of criticalTasks) {
    if (selected.find(s => s.id === task.id)) {
      score += 200
    } else {
      score -= 300
    }
  }

  // 2. –¶–µ–Ω–Ω–æ—Å—Ç—å –≤—ã–±—Ä–∞–Ω–Ω—ã—Ö –∑–∞–¥–∞—á
  score += selected.reduce((sum, t) => sum + t.hidden_value, 0)

  // 3. –≠—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç—å capacity (–∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–æ 90-100% = –±–æ–Ω—É—Å)
  const usageRatio = totalSP / scenario.sprintCapacity
  if (usageRatio >= 0.9 && usageRatio <= 1.0) score += 150
  else if (usageRatio >= 0.8) score += 75

  // 4. –ë–∞–ª–∞–Ω—Å —Ç–∏–ø–æ–≤ (—Ö–æ—Ç—è –±—ã 1 tech-debt = –±–æ–Ω—É—Å)
  const hasDebt = selected.some(t => t.type === 'tech-debt')

  // 5. –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –Ω–µ –Ω–∞—Ä—É—à–µ–Ω—ã
  for (const task of selected) {
    if (task.dependency && !selected.find(s => s.id === task.dependency)) {
      score -= 100
    }
  }

  return Math.max(0, score)
}
